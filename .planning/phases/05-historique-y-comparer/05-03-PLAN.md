Dentro del método `Historique.init()` existente (creado en 05-02), añadir la lógica del botón "Effacer tout" DESPUÉS del listener del botón CTA y ANTES de `Historique.render()`.

Insertar el siguiente bloque dentro de `Historique.init: function()`:

```javascript
// Botón Effacer tout — flujo 2 pasos con timeout 3s
var clearBtn = document.getElementById('btn-clear-history');
var clearTimer = null;
var clearStep = 0;  // 0 = normal, 1 = esperando confirmación

if (clearBtn) {
  clearBtn.addEventListener('click', function() {
    if (clearStep === 0) {
      // Paso 1: cambiar a estado "confirmar"
      clearStep = 1;
      clearBtn.textContent = I18n.t('historique.confirm_clear');
      clearBtn.className = 'btn-danger';

      // Timeout 3s: reset automático si no hay segundo clic
      clearTimer = setTimeout(function() {
        clearStep = 0;
        clearBtn.textContent = I18n.t('historique.clear');
        clearBtn.className = 'btn-outline-sm';
        clearTimer = null;
      }, 3000);

    } else if (clearStep === 1) {
      // Paso 2: confirmar borrado
      clearTimeout(clearTimer);
      clearTimer = null;
      clearStep = 0;
      History.clear();
      clearBtn.textContent = I18n.t('historique.clear');
      clearBtn.className = 'btn-outline-sm';
      Historique.render();
    }
  });
}
```
