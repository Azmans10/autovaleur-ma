```javascript
remove: function(id) {
  var all = History.getAll();
  var filtered = all.filter(function(entry) {
    return entry.id !== id;
  });
  try {
    localStorage.setItem(History._KEY, JSON.stringify(filtered));
  } catch (e) {
    console.warn('AutoValeur — no se pudo actualizar historial:', e.message);
  }
  return filtered;
},

clear: function() {
  try {
    localStorage.setItem(History._KEY, JSON.stringify([]));
  } catch (e) {
    console.warn('AutoValeur — no se pudo limpiar historial:', e.message);
  }
  return [];
},
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01-01 | Tampering | `History.getAll()` → `History.remove()` | mitigate | `getAll()` ya tiene try/catch que retorna `[]` en JSON inválido; `remove()` llama a `getAll()` y opera sobre el resultado — si está corrompido, filter retorna `[]` y se sobreescribe con array vacío |
| T-05-01-02 | Tampering | `History.remove(id)` — id recibido de botón DOM | accept | El id es un timestamp numérico generado por la propia app (`Date.now()`); no se evalúa ni se interpreta como código; se compara con `!==` |
| T-05-01-03 | Denial of Service | localStorage lleno (cuota excedida) | mitigate | `try/catch` en `remove()` y `clear()` captura `QuotaExceededError`; `console.warn` silencioso — patrón establecido en Fase 1-4 |
