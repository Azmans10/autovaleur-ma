**Paso C — Añadir módulo var Comparer en el bloque JS:**

Añadir después del módulo `var Historique` y antes del bloque DOMContentLoaded:

```javascript
// === Módulo Comparer ===
var Comparer = {
  init: function() {
    // Botón CTA del estado vacío
    var ctaBtn = document.getElementById('btn-comparer-cta');
    if (ctaBtn) {
      ctaBtn.addEventListener('click', function() {
        Router.navigate('estimer');
      });
    }

    // Listeners de los selects — renderizado automático
    var selectA = document.getElementById('select-car-a');
    var selectB = document.getElementById('select-car-b');

    if (selectA) {
      selectA.addEventListener('change', function() { Comparer._onSelectChange(); });
    }
    if (selectB) {
      selectB.addEventListener('change', function() { Comparer._onSelectChange(); });
    }

    // Render inicial
    Comparer.render();
  },

  render: function() {
    var entries = History.getAll();
    var emptyDiv = document.getElementById('comparer-empty');
    var controlsDiv = document.getElementById('comparer-controls');
    var resultDiv = document.getElementById('comparer-result');

    if (!emptyDiv || !controlsDiv) return;

    if (entries.length < 2) {
      // Menos de 2 valoraciones — mostrar mensaje orientativo
      emptyDiv.hidden = false;
      controlsDiv.hidden = true;
      if (resultDiv) resultDiv.innerHTML = '';
      return;
    }

    // 2 o más valoraciones — mostrar selects
    emptyDiv.hidden = true;
    controlsDiv.hidden = false;

    // Re-poblar los selects (Pitfall 7 prevenido: siempre re-poblar al render)
    var selectA = document.getElementById('select-car-a');
    var selectB = document.getElementById('select-car-b');

    if (!selectA || !selectB) return;

    var prevA = selectA.value;
    var prevB = selectB.value;

    // Limpiar opciones excepto placeholder
    selectA.innerHTML = '<option value=""></option>';
    selectB.innerHTML = '<option value=""></option>';

    entries.forEach(function(entry) {
      var label = entry.make + ' ' + entry.model + ' ' + entry.year + ' — ' + formatPrice(entry.estimated_price, I18n.currentLang);

      var optA = document.createElement('option');
      optA.value = entry.id;
      optA.textContent = label;  // textContent — XSS prevention
      selectA.appendChild(optA);

      var optB = document.createElement('option');
      optB.value = entry.id;
      optB.textContent = label;  // textContent — XSS prevention
      selectB.appendChild(optB);
    });

    // Restaurar selecciones previas si los ids siguen válidos
    if (prevA && selectA.querySelector('option[value="' + prevA + '"]')) {
      selectA.value = prevA;
    }
    if (prevB && selectB.querySelector('option[value="' + prevB + '"]')) {
      selectB.value = prevB;
    }

    // Re-renderizar comparación si ambos selects tienen valor válido
    Comparer._onSelectChange();
  },

  _onSelectChange: function() {
    var selectA = document.getElementById('select-car-a');
    var selectB = document.getElementById('select-car-b');
    var resultDiv = document.getElementById('comparer-result');

    if (!selectA || !selectB || !resultDiv) return;

    var idA = selectA.value;
    var idB = selectB.value;

    // Validar: ambos deben tener valor y ser distintos
    if (!idA || !idB || idA === idB) {
      resultDiv.innerHTML = '';
      return;
    }

    var entries = History.getAll();
    var entryA = null;
    var entryB = null;

    entries.forEach(function(e) {
      if (String(e.id) === String(idA)) entryA = e;
      if (String(e.id) === String(idB)) entryB = e;
    });

    if (!entryA || !entryB) {
      resultDiv.innerHTML = '';
      return;
    }

    Comparer.renderComparison(entryA, entryB);
  },

  renderComparison: function(entryA, entryB) {
    // La lógica de barras y ganador se implementa en 05-05
    // En este plan: renderizar el layout base de las dos tarjetas sin barras ni ganador
    var resultDiv = document.getElementById('comparer-result');
    if (!resultDiv) return;

    // Limpiar resultado anterior
    resultDiv.innerHTML = '';

    var grid = document.createElement('div');
    grid.className = 'compare-grid';

    [entryA, entryB].forEach(function(entry, index) {
      var card = document.createElement('div');
      card.className = 'compare-card';
      card.dataset.entryId = entry.id;

      var label = document.createElement('p');
      label.className = 'compare-card__label';
      label.textContent = index === 0 ? 'A' : 'B';

      var title = document.createElement('p');
      title.className = 'compare-card__title';
      title.textContent = entry.make + ' ' + entry.model + ' ' + entry.year;

      var price = document.createElement('p');
      price.className = 'compare-card__price';
      price.textContent = formatPrice(entry.estimated_price, I18n.currentLang);

      var meta = document.createElement('p');
      meta.className = 'compare-card__meta';
      meta.textContent = new Intl.NumberFormat('fr-FR').format(entry.mileage) + ' km';

      // Placeholder para barra de precio (llenado en 05-05)
      var barTrack = document.createElement('div');
      barTrack.className = 'compare-bar-track';
      var bar = document.createElement('div');
      bar.className = 'compare-bar';
      bar.dataset.price = entry.estimated_price;
      bar.style.width = '0%';  // Se calculará en 05-05
      barTrack.appendChild(bar);

      card.appendChild(label);
      card.appendChild(title);
      card.appendChild(price);
      card.appendChild(meta);
      card.appendChild(barTrack);
      grid.appendChild(card);
    });

    resultDiv.appendChild(grid);
  }
};
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| localStorage → select options | Datos de usuario (make, model, year, precio) se usan como texto de opciones del select |
| select.value → History.getAll().find | El id numérico viaja por el DOM como string |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-04-01 | Injection (XSS) | `entry.make + entry.model` → option text | mitigate | `optA.textContent = label` — textContent escapa HTML automáticamente; `label` contiene datos de usuario pero se inserta via textContent, nunca innerHTML |
| T-05-04-02 | Tampering | `select.value` como id → lookup en History | accept | El value del select es el id numérico generado por `Date.now()` en la propia app; `String(e.id) === String(idA)` hace comparación segura de strings |
| T-05-04-03 | Spoofing | Opciones de selects desincronizadas con historial real | mitigate | Pitfall 7 prevenido: `Comparer.render()` re-pobla ambos selects en cada hashchange → el historial DOM siempre refleja el localStorage actual |
</threat_model>
