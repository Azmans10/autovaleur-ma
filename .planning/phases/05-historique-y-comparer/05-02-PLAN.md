```javascript
// === Utilidad: fecha relativa ===
function formatRelativeDate(isoString) {
  var days = Math.floor((Date.now() - new Date(isoString).getTime()) / 86400000);
  if (days === 0) return I18n.t('date.today');
  if (days < 7)  return I18n.t('date.days_ago').replace('{n}', days);
  var weeks = Math.floor(days / 7);
  return I18n.t('date.weeks_ago').replace('{n}', weeks);
}

// === Módulo Historique ===
var Historique = {
  init: function() {
    // Botón CTA del empty state
    var ctaBtn = document.getElementById('btn-history-cta');
    if (ctaBtn) {
      ctaBtn.addEventListener('click', function() {
        Router.navigate('estimer');
      });
    }
    // Renderizado inicial
    Historique.render();
  },

  render: function() {
    var entries = History.getAll();
    var container = document.getElementById('history-list');
    var emptyState = document.getElementById('history-empty');
    var clearBtn = document.getElementById('btn-clear-history');

    if (!container) return;

    // Limpiar contenedor (sin datos de usuario en innerHTML — seguro)
    container.innerHTML = '';

    if (entries.length === 0) {
      emptyState.hidden = false;
      container.hidden = true;
      if (clearBtn) clearBtn.hidden = true;
      return;
    }

    emptyState.hidden = true;
    container.hidden = false;
    if (clearBtn) clearBtn.hidden = false;

    entries.forEach(function(entry) {
      var card = document.createElement('div');
      card.className = 'history-card card';
      card.setAttribute('role', 'listitem');

      // Header: título + botón eliminar
      var header = document.createElement('div');
      header.className = 'history-card__header';

      // Título: Marca Modelo Año (textContent — XSS prevention)
      var title = document.createElement('p');
      title.className = 'history-card__title';
      title.textContent = entry.make + ' ' + entry.model + ' ' + entry.year;

      // Botón eliminar ×
      var deleteBtn = document.createElement('button');
      deleteBtn.className = 'btn-delete';
      deleteBtn.setAttribute('aria-label', I18n.t('historique.delete'));
      deleteBtn.textContent = '×';
      deleteBtn.dataset.id = entry.id;
      deleteBtn.addEventListener('click', function() {
        History.remove(Number(this.dataset.id));
        Historique.render();
      });

      header.appendChild(title);
      header.appendChild(deleteBtn);

      // Precio destacado
      var price = document.createElement('p');
      price.className = 'history-card__price';
      price.textContent = formatPrice(entry.estimated_price, I18n.currentLang);

      // Meta: km + fecha relativa
      var meta = document.createElement('p');
      meta.className = 'history-card__meta';
      var kmText = new Intl.NumberFormat('fr-FR').format(entry.mileage) + ' km';
      var dateText = formatRelativeDate(entry.date);
      meta.textContent = kmText + ' · ' + dateText;

      card.appendChild(header);
      card.appendChild(price);
      card.appendChild(meta);
      container.appendChild(card);
    });
  }
};
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-02-01 | Injection (XSS) | `entry.make`, `entry.model`, `entry.city` → DOM | mitigate | Usar `element.textContent = entry.make` — NUNCA `innerHTML` con datos de usuario. `container.innerHTML = ''` (limpiar contenedor sin datos) es seguro. |
