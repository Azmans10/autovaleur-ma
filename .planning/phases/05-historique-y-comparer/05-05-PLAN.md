**Paso A — Añadir la función `determineWinner` como función standalone (no método de Comparer):**

Añadir en el bloque JS, justo ANTES de `var Comparer = {` (para que esté disponible cuando Comparer la llame):

```javascript
// === Función: determinar ganador por ratio precio/km ===
function determineWinner(entryA, entryB) {
  // Si alguno tiene mileage === 0: gana el de precio menor
  if (entryA.mileage === 0 || entryB.mileage === 0) {
    if (entryA.estimated_price < entryB.estimated_price) return entryA;
    if (entryB.estimated_price < entryA.estimated_price) return entryB;
    return null; // empate
  }
  // Ratio precio/km — menor es mejor
  var ratioA = entryA.estimated_price / entryA.mileage;
  var ratioB = entryB.estimated_price / entryB.mileage;
  if (ratioA < ratioB) return entryA;
  if (ratioB < ratioA) return entryB;
  return null; // empate
}
```

**Paso B — Actualizar `Comparer.renderComparison` para calcular barras y aplicar ganador:**

Localizar el método `renderComparison: function(entryA, entryB)` dentro de `var Comparer` (creado en 05-04) y REEMPLAZARLO COMPLETO con:

```javascript
renderComparison: function(entryA, entryB) {
  var resultDiv = document.getElementById('comparer-result');
  if (!resultDiv) return;

  // Limpiar resultado anterior
  resultDiv.innerHTML = '';

  // Calcular barras proporcionales
  var maxPrice = Math.max(entryA.estimated_price, entryB.estimated_price);
  var widthA = maxPrice > 0 ? (entryA.estimated_price / maxPrice * 100).toFixed(2) : 0;
  var widthB = maxPrice > 0 ? (entryB.estimated_price / maxPrice * 100).toFixed(2) : 0;

  // Determinar ganador
  var winner = determineWinner(entryA, entryB);

  var grid = document.createElement('div');
  grid.className = 'compare-grid';

  [[entryA, widthA], [entryB, widthB]].forEach(function(pair, index) {
    var entry = pair[0];
    var barWidth = pair[1];
    var isWinner = winner && winner.id === entry.id;

    var card = document.createElement('div');
    card.className = isWinner ? 'compare-card compare-winner' : 'compare-card';
    card.dataset.entryId = entry.id;

    // Badge "Meilleur rapport" — solo para el ganador
    if (isWinner) {
      var badge = document.createElement('span');
      badge.className = 'badge';
      badge.textContent = I18n.t('comparer.winner_badge');
      card.appendChild(badge);
    }

    // Label A / B
    var label = document.createElement('p');
    label.className = 'compare-card__label';
    label.textContent = index === 0 ? 'A' : 'B';

    // Título: Marca Modelo Año (textContent — XSS prevention)
    var title = document.createElement('p');
    title.className = 'compare-card__title';
    title.textContent = entry.make + ' ' + entry.model + ' ' + entry.year;

    // Precio destacado
    var price = document.createElement('p');
    price.className = 'compare-card__price';
    price.textContent = formatPrice(entry.estimated_price, I18n.currentLang);

    // Meta: km
    var meta = document.createElement('p');
    meta.className = 'compare-card__meta';
    meta.textContent = new Intl.NumberFormat('fr-FR').format(entry.mileage) + ' km';

    // Barra de precio proporcional
    var barLabel = document.createElement('p');
    barLabel.className = 'compare-card__meta';
    barLabel.style.marginBlockEnd = '0';
    // Etiqueta de la barra no necesita texto — la barra es visual

    var barTrack = document.createElement('div');
    barTrack.className = 'compare-bar-track';

    var bar = document.createElement('div');
    bar.className = 'compare-bar';
    // Forzar reflow para que la animación CSS (transition) se dispare correctamente
    // Primero width: 0%, luego en el siguiente frame el width real
    bar.style.width = '0%';
    barTrack.appendChild(bar);

    card.appendChild(label);
    card.appendChild(title);
    card.appendChild(price);
    card.appendChild(meta);
    card.appendChild(barTrack);
    grid.appendChild(card);

    // Animar la barra al frame siguiente (permite que la transición CSS se active)
    (function(b, w) {
      requestAnimationFrame(function() {
        b.style.width = w + '%';
      });
    })(bar, barWidth);
  });

  resultDiv.appendChild(grid);
}
```
