  <action>
Insertar el modulo I18n en el bloque `<script>` de index.html, DESPUES del objeto STRINGS (insertado por Plan 01) y ANTES de la seccion ROUTES/Router. Tambien modificar el handler DOMContentLoaded para llamar a I18n.init() ANTES de Router.init().

1. Insertar el siguiente modulo DESPUES de STRINGS y ANTES de ROUTES:

```javascript
// ===============================================================
// I18N — Motor de traduccion FR / AR
// Lee STRINGS, aplica textContent/innerHTML/aria-label segun
// atributos data-i18n, data-i18n-html, data-i18n-aria.
// ===============================================================

var I18n = {
  currentLang: 'fr',

  /**
   * Obtener la traduccion de una clave.
   * Devuelve la clave misma si no se encuentra (fail visible).
   */
  t: function(key) {
    var strings = STRINGS[I18n.currentLang];
    return (strings && strings[key]) || key;
  },

  /**
   * Cambiar el idioma activo.
   * Actualiza dir, lang, localStorage, y reaplica traducciones.
   */
  setLang: function(lang) {
    if (!STRINGS[lang]) return;
    I18n.currentLang = lang;
    document.documentElement.lang = lang;
    document.documentElement.dir = (lang === 'ar') ? 'rtl' : 'ltr';
    localStorage.setItem('lang', lang);
    I18n._applyTranslations();
    // Actualizar pill toggle si existe (Plan 03)
    document.querySelectorAll('.lang-option').forEach(function(opt) {
      opt.classList.toggle('active', opt.getAttribute('data-lang') === lang);
    });
  },

  /**
   * Recorrer el DOM y aplicar traducciones segun atributos data-i18n.
   */
  _applyTranslations: function() {
    // Texto interior (textContent o innerHTML)
    document.querySelectorAll('[data-i18n]').forEach(function(el) {
      var key = el.getAttribute('data-i18n');
      var val = I18n.t(key);
      if (el.hasAttribute('data-i18n-html')) {
        // Banner iOS: reemplazar {icon} con el SVG inline
        var svgIcon = '<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" style="vertical-align:middle" aria-hidden="true"><path d="M4 12v8a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2v-8"/><polyline points="16 6 12 2 8 6"/><line x1="12" y1="2" x2="12" y2="15"/></svg>';
        el.innerHTML = val.replace('{icon}', svgIcon);
      } else {
        el.textContent = val;
      }
    });
    // aria-label
    document.querySelectorAll('[data-i18n-aria]').forEach(function(el) {
      var key = el.getAttribute('data-i18n-aria');
      el.setAttribute('aria-label', I18n.t(key));
    });
  },

  /**
   * Inicializar: leer preferencia de localStorage, aplicar idioma.
   * Llamar ANTES de Router.init() en DOMContentLoaded.
   */
  init: function() {
    var saved = localStorage.getItem('lang');
    // Solo aceptar 'fr' o 'ar'; cualquier otro valor → default 'fr'
    var lang = (saved === 'ar') ? 'ar' : 'fr';
    I18n.setLang(lang);
  }
};
```
