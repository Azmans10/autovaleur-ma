```javascript
    // ===============================================================
    // ESTIMER FORM — Formulario de valoración
    // Cascada: marca → modelo → año desde CAR_DB
    // Submit handler se añade en plan 04-02
    // ===============================================================
    var EstimerForm = {

      /**
       * Poblar el select de marcas con las claves de CAR_DB (ordenadas).
       * Llamado una vez al init.
       */
      _populateMakes: function() {
        var makeSelect = document.getElementById('field-make');
        if (!makeSelect) return;
        // Conservar la option vacía inicial (primer hijo)
        var placeholder = makeSelect.querySelector('option[value=""]');
        makeSelect.innerHTML = '';
        if (placeholder) makeSelect.appendChild(placeholder);
        Object.keys(CAR_DB).sort().forEach(function(make) {
          var opt = document.createElement('option');
          opt.value = make;
          opt.textContent = make;
          makeSelect.appendChild(opt);
        });
      },

      /**
       * Poblar el select de modelos para una marca dada.
       * Si make es vacío/nulo, deshabilita el select y limpia las opciones.
       * @param {string} make
       */
      _populateModels: function(make) {
        var modelSelect = document.getElementById('field-model');
        if (!modelSelect) return;
        var placeholder = modelSelect.querySelector('option[value=""]');
        modelSelect.innerHTML = '';
        if (placeholder) modelSelect.appendChild(placeholder);
        modelSelect.disabled = !make;
        modelSelect.value = '';
        if (!make || !CAR_DB[make]) return;
        Object.keys(CAR_DB[make]).sort().forEach(function(model) {
          var opt = document.createElement('option');
          opt.value = model;
          opt.textContent = model;
          modelSelect.appendChild(opt);
        });
      },

      /**
       * Poblar el select de años para marca+modelo dados.
       * Orden descendente: año más reciente primero.
       * Si make o model están vacíos, deshabilita el select.
       * @param {string} make
       * @param {string} model
       */
      _populateYears: function(make, model) {
        var yearSelect = document.getElementById('field-year');
        if (!yearSelect) return;
        var placeholder = yearSelect.querySelector('option[value=""]');
        yearSelect.innerHTML = '';
        if (placeholder) yearSelect.appendChild(placeholder);
        yearSelect.disabled = !(make && model);
        yearSelect.value = '';
        if (!make || !model || !CAR_DB[make] || !CAR_DB[make][model]) return;
        var range = CAR_DB[make][model].years; // [minYear, maxYear]
        for (var y = range[1]; y >= range[0]; y--) {
          var opt = document.createElement('option');
          opt.value = String(y);
          opt.textContent = String(y);
          yearSelect.appendChild(opt);
        }
      },

      /**
       * Recalcular el estado del botón submit.
       * Disabled si marca, modelo o año están sin seleccionar.
       */
      _updateSubmitState: function() {
        var make  = document.getElementById('field-make').value;
        var model = document.getElementById('field-model').value;
        var year  = document.getElementById('field-year').value;
        var btn   = document.getElementById('btn-estimer');
        if (btn) btn.disabled = !(make && model && year);
      },

      /**
       * Inicializar el formulario: poblar marcas, conectar listeners de cascada.
       * Llamado desde DOMContentLoaded.
       */
      init: function() {
        EstimerForm._populateMakes();

        var makeSelect  = document.getElementById('field-make');
        var modelSelect = document.getElementById('field-model');

        if (makeSelect) {
          makeSelect.addEventListener('change', function() {
            var make = makeSelect.value;
            EstimerForm._populateModels(make);
            EstimerForm._populateYears('', '');
            EstimerForm._updateSubmitState();
          });
        }

        if (modelSelect) {
          modelSelect.addEventListener('change', function() {
            var make  = document.getElementById('field-make').value;
            var model = modelSelect.value;
            EstimerForm._populateYears(make, model);
            EstimerForm._updateSubmitState();
          });
        }

        var yearSelect = document.getElementById('field-year');
        if (yearSelect) {
          yearSelect.addEventListener('change', function() {
            EstimerForm._updateSubmitState();
          });
        }

        // Estado inicial del botón (disabled por defecto, ya está en HTML)
        EstimerForm._updateSubmitState();
      }
    };
```
