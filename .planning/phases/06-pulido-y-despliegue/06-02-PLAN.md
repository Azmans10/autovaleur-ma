**Bloque A — Modulo ErrorUI (insertar en la seccion JS, despues del modulo History y antes de los event listeners o del modulo que invoque History):**

```javascript
/* ── ErrorUI ────────────────────────────────────────────── */
var ErrorUI = {
  showStorageBanner: function() {
    var banner = document.getElementById('error-storage-banner');
    if (banner) {
      banner.hidden = false;
      I18n.update(); // actualizar las cadenas del banner al idioma actual
    }
  },
  closeBanner: function() {
    var banner = document.getElementById('error-storage-banner');
    if (banner) { banner.hidden = true; }
  },
  showInline: function(containerId, i18nKey) {
    var container = document.getElementById(containerId);
    if (!container) return;
    // Eliminar error previo si existe
    var prev = container.querySelector('.error-inline');
    if (prev) prev.remove();
    var p = document.createElement('p');
    p.className = 'error-inline';
    p.setAttribute('data-i18n', i18nKey);
    p.textContent = I18n.t(i18nKey);
    container.appendChild(p);
  },
  clearInline: function(containerId) {
    var container = document.getElementById(containerId);
    if (!container) return;
    var prev = container.querySelector('.error-inline');
    if (prev) prev.remove();
  }
};
```
